print('Задание №1')
x = int(input('Введите число'))
if x<0:
    if x%2==0:
        print('Число отрицательное четное')
    else:
        print('Число отрицательное нечетное')
else:
    if x==0:
        print('Число равно нулю')
    else:
        if x%2==0:
            print('Число положительное четное')
        else:
            print('Число положительное нечетное')
